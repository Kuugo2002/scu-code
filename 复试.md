## 判断一个字符是什么类型
```cpp
#include <cctype>
char ch='A';
isalnum(ch); //判断是否是字母或者数字
isalpha(ch); //判断是否是字母
isdigit(ch); //判断是否是数字
islower(ch); //判断是否是小写字母
isupper(ch); //判断是否是大写字母
isspace(ch); //判断是否是空白字符
```

## string的find函数
```cpp
#include<iostream>
#include<string>
int main()
{
    string str="Hello world, Hello C++";
    // 查找子串
    size_t pos=str.find("Hello");
    if(pos!=string::npos){
        cout << "找到'Hello'在位置: " << pos << endl;  // 输出: 0
    }
    // 从指定位置开始查找
    pos=str.find("Hello",5); // 从索引5开始查找
    if(pos!=string::npos){
        cout << "从位置5开始找到'Hello'在: " << pos << endl;  // 输出: 13
    }
    // 查找字符
    pos=str.find('o');
    if(pos!=string::npos){
        cout << "找到字符'o'在位置: " << pos << endl;  // 输出: 4
    }
}
```
